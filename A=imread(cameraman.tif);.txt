%otsu
[h,~]=imhist(A);
h=h(:)';
total=r*c;
p=h/total;
n=0:255;
max_var=0;
thr=0;
for t = 0:255
    back = p(1:t+1);
    fore = p(t+2:256);
    wb = sum(back);
    wf = sum(fore);
    if wb == 0 || wf==0
        continue;
    end
    mean_b = sum(n(1:t+1).*back)/wb;
    mean_f = sum(n(t+2:256).*fore)/wf;
    var_between = wf*wb*(mean_b-mean_f)^2;
    if var_between > max_var
        max_var=var_between;
        thr=t;
    end
end
b=A>thr;
subplot(5,3,12);
imshow(b);
title(['otsu th = ', num2str(thr)]);

J=zeros(size(A),"uint8");
gx=[-1 0 1; -2 0 2 ;-1 0 1];
gy=[-1 -2 -1; 0 0 0; 1 2 1];
for r = 2:row-1
    for c = 2:col-1
        patch = A(r-1:r+1,c-1:c+1);
        hx = gx.*patch;
        hy = gy.*patch;

        kx = sum(hx,"all");
        ky = sum(hy,"all");

        J(r,c)=sqrt((kx^2 + ky^2));
    end 
end
